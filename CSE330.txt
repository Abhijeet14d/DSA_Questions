Q: find the number is prime or not?

#include <iostream>
#include <cmath>
using namespace  std;

bool isprime(int n){
  if(n<=1){
    return false;
  };
  for(int i=2;i<sqrt(n);i++){
    if(n%i==0){
      return false;
    }
  }
  return true; 
}

bool isprimeOptimize(int n){
  if(n<=1){
    return false;
  }
  if(n<=3){
    return true;
  }
  if(n%2==0 || n%3==0){
    return false;
  }
  for(int i=5;i*i<=sqrt(n);i=i+6){
    if(n%i==0 || n%(i+2)==0){
      return false;
    }
  }
  return true;
}

int main(){
  int n;
  cout << "Enter the number: ";
  cin >> n;

  bool x = isprimeOptimize(n);
  if(x) cout << "is prime"<< endl;
  else cout << "not prime"<< endl;
}
